178 // Helper looks up the bytecode for the specified vertex shader permutation.
179 // Client code needs this in order to create matching input layouts.
180 void GetVertexShaderBytecode(int permutation, _Out_ void const** pShaderByteCode, _Out_ size_t* pByteCodeLength) noexcept
181 {
182 assert(pShaderByteCode != nullptr && pByteCodeLength != nullptr);
183 assert(permutation >= 0 && permutation < Traits::ShaderPermutationCount);
184 _Analysis_assume_(permutation >= 0 && permutation < Traits::ShaderPermutationCount);
185 const int shaderIndex = VertexShaderIndices[permutation];
186 assert(shaderIndex >= 0 && shaderIndex < Traits::VertexShaderCount);
187 _Analysis_assume_(shaderIndex >= 0 && shaderIndex < Traits::VertexShaderCount);
188
189 ShaderBytecode const& bytecode = VertexShaderBytecode[shaderIndex];
190
191 *pShaderByteCode = bytecode.code;
192 *pByteCodeLength = bytecode.length;
193 }

196 // Helper sets our shaders and constant buffers onto the D3D device.
197 void ApplyShaders(_In_ ID3D11DeviceContext* deviceContext, int permutation)
198 {
199 // Set shaders.
200 auto vertexShader = mDeviceResources->GetVertexShader(permutation);
201 auto pixelShader = mDeviceResources->GetPixelShader(permutation);
202
203 deviceContext->VSSetShader(vertexShader, nullptr, 0);
204 deviceContext->PSSetShader(pixelShader, nullptr, 0);
205
206 #if defined(_XBOX_ONE) && defined(_TITLE)
207 void *grfxMemory;
208 mConstantBuffer.SetData(deviceContext, constants, &grfxMemory);
209
210 Microsoft::WRL::ComPtr<ID3D11DeviceContextX> deviceContextX;
211 ThrowIfFailed(deviceContext->QueryInterface(IID_GRAPHICS_PPV_ARGS(deviceContextX.GetAddressOf())));
212
213 auto buffer = mConstantBuffer.GetBuffer();
214
215 deviceContextX->VSSetPlacementConstantBuffer(0, buffer, grfxMemory);
216 deviceContextX->PSSetPlacementConstantBuffer(0, buffer, grfxMemory);
217 #else
218 // Make sure the constant buffer is up to date.
219 if (dirtyFlags & EffectDirtyFlags::ConstantBuffer)
220 {
221 mConstantBuffer.SetData(deviceContext, constants);
222
223 dirtyFlags &= ~EffectDirtyFlags::ConstantBuffer;
224 }
225
226 // Set the constant buffer.
227 ID3D11Buffer* buffer = mConstantBuffer.GetBuffer();
228
229 deviceContext->VSSetConstantBuffers(0, 1, &buffer);
230 deviceContext->PSSetConstantBuffers(0, 1, &buffer);
231 #endif
232 }

264 // Gets or lazily creates the specified vertex shader permutation.
265 ID3D11VertexShader* GetVertexShader(int permutation)
266 {

268 _Analysis_assume_(permutation >= 0 && permutation < Traits::ShaderPermutationCount);
