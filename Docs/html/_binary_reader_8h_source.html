33 // Reads a single value.
34 template<typename T> T const& Read()
35 {
36 return *ReadArray<T>(1);
37 }

40 // Reads an array of values.
41 template<typename T> T const* ReadArray(size_t elementCount)
42 {
43 static_assert(std::is_standard_layout<T>::value, "Can only read plain-old-data types");
44
45 uint8_t const* newPos = mPos + sizeof(T) * elementCount;
46
47 if (newPos < mPos)
48 throw std::overflow_error("ReadArray");
49
50 if (newPos > mEnd)
51 throw std::runtime_error("End of file");
52
53 auto result = reinterpret_cast<T const*>(mPos);
54
55 mPos = newPos;
56
57 return result;
58 }
